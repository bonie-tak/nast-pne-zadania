var D1_kub = prompt("Podaj pierwszy dzień (D1):", "");
var G1_kub = prompt("Podaj pierwszą godzinę (G1):", "");
var D2_kub = prompt("Podaj drugi dzień (D2):", "");
var G2_kub = prompt("Podaj drugą godzinę (G2):", "");

var d1 = parseInt(D1_kub);
var g1 = parseInt(G1_kub);
var d2 = parseInt(D2_kub);
var g2 = parseInt(G2_kub);

// Obliczanie różnicy w sekundach
var roznica_dni = (d1 - d2) * 24 * 3600;
var roznica_godzin = (g1 - g2) * 3600;
var sekundy = roznica_dni + roznica_godzin;

// Zamiana na wartość dodatnią
if (sekundy < 0) sekundy = -sekundy;

document.write("D1 = " + d1 + ", G1 = " + g1 + "<br>");
document.write("D2 = " + d2 + ", G2 = " + g2 + "<br>");
document.write("Ilość sekund = " + sekundy + " sekund");
