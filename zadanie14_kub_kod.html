<script>
function rozkladUlamek() {

    var m = 4;

        "1/" + m + " = 1/6 + 1/30 + 1/20<br><br>" +
        "Sprawdzenie: 1/6 + 1/30 + 1/20 = " + suma.toFixed(3);
}
